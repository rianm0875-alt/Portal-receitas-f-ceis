function renderCategories() {
  categoryGrid.innerHTML = categories
    .map(([name, text, examples]) => `
      <article class="category-card">
        <button class="category-title-button" type="button" data-category="${name}">${name}</button>
        <span>${text}</span>
        <div class="category-actions">
          ${examples.map((example) => `<button type="button" data-category="${name}" data-topic="${example}">${example}</button>`).join("")}
        </div>
      </article>
    `)
    .join("");

  categoryGrid.querySelectorAll("[data-category]").forEach((button) => {
    button.addEventListener("click", () => {
      applyCategorySearch(button.dataset.category, button.dataset.topic || "");
    });
  });
}

function applyCategorySearch(category, topic) {
  activeFilter = category;
  activeTopic = topic;
  activeTopicTerms = topic ? getTopicTerms(topic) : [];
  searchTerm = "";
  searchInput.value = topic;
  filters.forEach((item) => item.classList.toggle("active", item.dataset.filter === category));
  categoryGrid.querySelectorAll("[data-topic]").forEach((item) => item.classList.toggle("active", item.dataset.topic === topic));
  renderRecipes();
  document.querySelector("#receitas").scrollIntoView({ behavior: "smooth" });
}

function getTopicTerms(topic) {
  return (topicAliases[topic] || [topic]).map((term) => term.toLowerCase());
}

function renderDailyRecipe() {
  const today = new Date();
  const startOfYear = new Date(today.getFullYear(), 0, 0);
  const dayOfYear = Math.floor((today - startOfYear) / 86400000);
  dailyRecipe = recipes[dayOfYear % recipes.length];
  document.querySelector("#dailyImage").src = dailyRecipe.image;
  document.querySelector("#dailyImage").alt = dailyRecipe.title;
  document.querySelector("#dailyImage").onerror = () => {
    document.querySelector("#dailyImage").src = imageByCategory[dailyRecipe.category];
  };
  document.querySelector("#spotlight-title").textContent = dailyRecipe.title;
  document.querySelector("#dailyDescription").textContent = dailyRecipe.description;
  document.querySelector("#dailyMeta").textContent = `${dailyRecipe.time} | ${dailyRecipe.level}`;
}

        <strong>${recipe.title}</strong>
        <span>${recipe.category} | ${recipe.time}</span>
      </span>
    </button>
  `).join("");

  carouselTrack.querySelectorAll("[data-carousel-recipe]").forEach((button) => {
    button.addEventListener("click", () => openRecipe(recipes.find((recipe) => recipe.id === button.dataset.carouselRecipe)));
  });
}

function renderRecipes() {
  const normalizedTerm = searchTerm.trim().toLowerCase();
  const visibleRecipes = recipes.filter((recipe) => {
    const matchesFilter = activeFilter === "Todas" || recipe.category === activeFilter;
    const matchesTime =
      activeTime === "Todos" ||
      (activeTime === "Rapido" && recipe.timeMinutes <= 25) ||
      (activeTime === "Medio" && recipe.timeMinutes <= 45) ||
      (activeTime === "Longo" && recipe.timeMinutes > 45);
    const matchesLevel = activeLevel === "Todas" || recipe.level === activeLevel;
    const content = `${recipe.title} ${recipe.category} ${recipe.description} ${recipe.ingredients.join(" ")}`.toLowerCase();
    const matchesSearch = !normalizedTerm || content.includes(normalizedTerm);
    const matchesTopic = activeTopicTerms.length === 0 || activeTopicTerms.some((term) => content.includes(term));
    return matchesFilter && matchesTime && matchesLevel && matchesSearch && matchesTopic;
  });

  recipeCount.textContent = `${visibleRecipes.length} receita${visibleRecipes.length === 1 ? "" : "s"} encontrada${visibleRecipes.length === 1 ? "" : "s"}${activeTopic ? ` em ${activeTopic}` : ""}`;

  recipeGrid.innerHTML = visibleRecipes.length
    ? visibleRecipes.map((recipe) => `
      <article class="recipe-card">
        <img src="${recipe.image}" alt="${recipe.title}" onerror="this.src='${imageByCategory[recipe.category]}'">
        <div class="recipe-card-content">
          <p class="eyebrow">${recipe.category}</p>
          <h3>${recipe.title}</h3>
          <p>${recipe.description}</p>
          <div class="meta">
            <span>${recipe.time}</span>
            <span>${recipe.level}</span>
            <span>${recipe.servings}</span>
          </div>
          <div class="card-actions">
            <button type="button" data-recipe="${recipe.id}">Ver receita</button>
            <button class="favorite-button ${favorites.includes(recipe.id) ? "active" : ""}" type="button" data-favorite="${recipe.id}">
              ${favorites.includes(recipe.id) ? "Favorita" : "Favoritar"}
            </button>
          </div>
        </div>
      </article>
    `).join("")
    : `<p class="empty">Nenhuma receita encontrada. Tente outro termo ou categoria.</p>`;

  recipeGrid.querySelectorAll("[data-recipe]").forEach((button) => {
    button.addEventListener("click", () => openRecipe(recipes.find((recipe) => recipe.id === button.dataset.recipe)));
  });

  recipeGrid.querySelectorAll("[data-favorite]").forEach((button) => {
    button.addEventListener("click", () => toggleFavorite(button.dataset.favorite));
  });
}

function renderFavorites() {
  const favoriteRecipes = recipes.filter((recipe) => favorites.includes(recipe.id));
  favoritesList.innerHTML = favoriteRecipes.length
    ? favoriteRecipes.map((recipe) => `
      <article class="favorite-item">
        <strong>${recipe.title}</strong>
        <span>${recipe.category} | ${recipe.time} | ${recipe.level}</span>
      </article>
    `).join("")
    : `<p class="empty">Suas receitas favoritas vao aparecer aqui.</p>`;
}

function openRecipe(recipe) {
  if (!recipe) return;
  document.querySelector("#detailImage").src = recipe.image;
  document.querySelector("#detailImage").alt = recipe.title;
  document.querySelector("#detailImage").onerror = () => {
    document.querySelector("#detailImage").src = imageByCategory[recipe.category];
  };
  document.querySelector("#detailCategory").textContent = recipe.category;
  document.querySelector("#detailTitle").textContent = recipe.title;
  document.querySelector("#detailDescription").textContent = recipe.description;
  document.querySelector("#detailTime").textContent = recipe.time;
  document.querySelector("#detailLevel").textContent = recipe.level;
  document.querySelector("#detailServings").textContent = recipe.servings;
  document.querySelector("#detailIngredients").innerHTML = recipe.ingredients.map((item) => `<li>${item}</li>`).join("");
  document.querySelector("#detailSteps").innerHTML = recipe.steps.map((step) => `<li>${step}</li>`).join("");
  detail.classList.add("open");
}

function toggleFavorite(id) {
  favorites = favorites.includes(id) ? favorites.filter((favorite) => favorite !== id) : [...favorites, id];
  localStorage.setItem("portalReceitasFaceisFavoritas", JSON.stringify(favorites));
  renderRecipes();
  renderFavorites();
}
